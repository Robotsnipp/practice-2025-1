def update_state(self, txn, state):

    for key in txn:
        if key in state:

            state[key] = txn[key]
    return state

def is_valid_transaction(self, txn, state):
    if sum(txn.values()) != 0:
        return False
    for key in txn:
        balance = state.get(key, 0)
        if balance + txn[key] < 0:
            return False
    return True
